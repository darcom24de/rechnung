  <button onclick="downloadPDF()">PDF herunterladen</button>

  <script>
    async function downloadPDF() {
      const { jsPDF } = window.jspdf;
      const doc = new jsPDF();

      const kunde = document.getElementById('kunde').value;
      const beschreibung = document.getElementById('beschreibung').value;
      const betrag = document.getElementById('betrag').value;

      doc.setFontSize(18);
      doc.text("Rechnung", 20, 20);

      doc.setFontSize(12);
      doc.text(`Kunde: ${kunde}`, 20, 40);
      doc.text(`Beschreibung: ${beschreibung}`, 20, 50);
      doc.text(`Betrag: €${betrag}`, 20, 60);

      doc.text("Vielen Dank für Ihren Auftrag!", 20, 80);

      doc.save(`Rechnung_${kunde}.pdf`);
    }
  </script>
